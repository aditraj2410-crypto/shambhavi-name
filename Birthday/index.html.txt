/* ===================== MUSIC & START ===================== */
function startExperience(){
    document.getElementById("bgMusic").play();
    showScreen("s2");
}

/* ===================== SCREEN SWITCH ===================== */
function showScreen(id){
    document.querySelectorAll(".screen").forEach(s=>s.classList.remove("active"));
    document.getElementById(id).classList.add("active");
}

/* ===================== MEMORY SLIDESHOW ===================== */
let memIndex = 0;
const memories = [
 {type:"img",src:"assets/memory1.jpg",cap:""},
 {type:"img",src:"assets/memory2.jpg",cap:""},
 {type:"img",src:"assets/memory3.jpg",cap:"Pagli hai kya 😂😂"},
 {type:"video",src:"assets/memory.mp4"}
];

function nextMemory(){
    if(memIndex >= memories.length){
        showScreen("s5");
        return;
    }
    const box = document.getElementById("memoryBox");
    const cap = document.getElementById("memoryCaption");
    box.innerHTML=""; cap.innerHTML="";
    const m = memories[memIndex];
    if(m.type==="img"){
        const i=document.createElement("img");
        i.src=m.src;
        box.appendChild(i);
        cap.innerHTML=m.cap || "";
    }else{
        const v=document.createElement("video");
        v.src=m.src;
        v.autoplay=true;
        v.muted=true;
        v.playsInline=true;
        box.appendChild(v);
    }
    memIndex++;
}

/* ===================== OPTION LOGIC ===================== */
let seenFirst=false, seenMem=false;

function openFirstMessage(){
    seenFirst=true;
    showScreen("s6");
}
function openMemoriesText(){
    seenMem=true;
    showScreen("s7");
}
function nextAfterFirst(){
    if(!seenMem){ openMemoriesText(); }
    else{ showScreen("s8"); }
}
function nextAfterMemories(){
    if(!seenFirst){ openFirstMessage(); }
    else{ showScreen("s8"); }
}
